# Cloud Shell Deployment Commands
# Run these in Google Cloud Shell (https://console.cloud.google.com/ shell)

# 1. Configure Project
gcloud config set project sibuketu-83cc0

# 2. Build Docker Image
docker build -t gcr.io/sibuketu-83cc0/video-editor .

# 3. Push to Container Registry
docker push gcr.io/sibuketu-83cc0/video-editor

# 4. Deploy to Cloud Run
gcloud run deploy video-editor \
  --image gcr.io/sibuketu-83cc0/video-editor \
  --platform managed \
  --region asia-northeast1 \
  --allow-unauthenticated \
  --memory 2Gi \
  --cpu 2 \
  --timeout 300

# 5. Get Service URL
gcloud run services describe video-editor --region asia-northeast1 --format "value(status.url)"
